870 /*
871  * Searches the database for students from student names or usernames. Usernames
872  * are considered best matches and are come first in the returned array. If
873  * exactly two names are specified in the query, a first AND last name search is
874  * run, and these results are considered the next best match. Results matching
875  * first OR last name are given lowest priority and are returned last in the
876  * array.
877  * Returns an array of usernames (RIT DCEs)
878  */
879 function _hdata_student_search($query) {
880  if(!is_array($query)) {
881  $tokens = explode(' ',$query);
882  } else {
883  $tokens = $query;
884  }
885 
886  $DCEs = array();
887  $names = array();
888  foreach($tokens as $s) {
889  if(preg_match('/[a-zA-Z]{3}[0-9]{4}/', $s)) {
890  //definitely a student DCE
891  $DCEs[] = $s;
892  } else {
893  //not a DCE
894  $names[] = $s;
895  }
896  }
897 
898  /*Prioritize search results:
899  * 1. DCE match - top priority
900  * 2. First and last name match
901  * 3. One name match
902  */
903  $search_results = array();
904 
905  if(count($DCEs) > 0) {
906  $query = "SELECT p.username FROM {h_students_peoplesoft} p WHERE ";
907  foreach($DCEs as $s) {
908  $query .= "(p.username = '%s') OR ";
909  }
910  $query = substr($query, 0, -3); //trim the extra 'OR' from the end of the string
911  $query .= "UNION SELECT s.username FROM {h_students} s WHERE ";
912  foreach($DCEs as $s) {
913  $query .= "(s.username = '%s') OR ";
914  }
915  $query = substr($query, 0, -3); //trim the extra 'OR'
916  $rs = db_query($query, array_merge_recursive($DCEs,$DCEs));
917  while($arry = db_fetch_array($rs)) {
918  if(!in_array($arry['username'], $search_results)){
919  $search_results[] = $arry['username'];
920  }
921  }
922  }
923 
924  if(count($names) == 2) {
925  $query = "SELECT p.username username
926  FROM {h_students_peoplesoft} p
927  WHERE (p.firstname = '%s' AND p.lastname = '%s') OR
928  (p.firstname = '%s' AND p.lastname = '%s')
929  UNION
930  SELECT s.username username
931  FROM {h_students} s
932  WHERE (s.firstname = '%s' AND s.lastname = '%s') OR
933  (s.firstname = '%s' AND s.lastname = '%s')";
934  $result = db_query($query, $names[0], $names[1], $names[1], $names[0],
935  $names[0], $names[1], $names[1], $names[0]);
936  while($arry = db_fetch_array($result)) {
937  $search_results[] = $arry['username'];
938  }
939  }
940 
941  if(count($names) > 0) {
942  $query_names = '(';
943  foreach($names as $s) {
944  $query_names .= "'%s',";
945  }
946  $query_names = substr($query_names, 0, -1) . ')';
947 
948  $query = "SELECT username FROM {h_students_peoplesoft}
949  WHERE firstname IN $query_names OR lastname IN $query_names";
950  //using array_merge_recursive because we are using each name as a search
951  //criterion twice (firstname and lastname)
952  $rs = db_query($query, array_merge_recursive($names, $names));
953  while($arry = db_fetch_array($rs)) {
954  if(!in_array($arry['username'], $search_results)) {
955  $search_results[] = $arry['username'];
956  }
957  }
958 
959  $query = "SELECT username FROM {h_students}
960  WHERE firstname IN $query_names OR lastname IN $query_names";
961  $rs = db_query($query, array_merge_recursive($names,$names));
962  while($arry = db_fetch_array($rs)) {
963  if(!in_array($arry['username'], $search_results)) {
964  $search_results[] = $arry['username'];
965  }
966  }
967  }
968 
969  if(count($search_results) > 0) {
970  return $search_results;
971  } else {
972  return false;
973  }
